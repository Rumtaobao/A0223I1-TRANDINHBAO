<script>
    let arr = [1,3,5,6,9,8,7,2,10,4];

    function sortDescending(arr) {
        for (let i = 0; i < arr.length - 1; i++) {
            for (let j = i + 1; j < arr.length; j++) {
                if (arr[i] < arr[j]) {
                    let temp = arr[i];
                    arr[i] = arr[j];
                    arr[j] = temp;
                }
            }
        }
        return arr;
    }

    let sortedArr = sortDescending(arr);
    document.write(sortedArr);


</script>
